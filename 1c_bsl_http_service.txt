// Метод POST для /ping
Функция ОбработкаПинга(Запрос)
	
	Ответ = Новый HTTPСервисОтвет(200);
	Ответ.Заголовки.Вставить("Content-Type", "application/json; charset=utf-8");
	
	СтруктураОтвета = Новый Структура;
	СтруктураОтвета.Вставить("status", "success");
	СтруктураОтвета.Вставить("configuration", Метаданные.Синоним);
	СтруктураОтвета.Вставить("version", "8.3.24.1368");
	СтруктураОтвета.Вставить("serverTime", ТекущаяДатаСеанса());
	
	ЗаписьJSON = Новый ЗаписьJSON;
	ЗаписьJSON.УстановитьСтроку();
	ЗаписатьJSON(ЗаписьJSON, СтруктураОтвета);
	СтрокаJSON = ЗаписьJSON.Закрыть();
	
	Ответ.УстановитьТелоИзСтроки(СтрокаJSON, КодировкаТекста.UTF8, ИспользованиеByteOrderMark.НеИспользовать);
	Возврат Ответ;
	
КонецФункции
